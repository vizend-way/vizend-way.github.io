Getting Started

Vizend 드라마 개발에 항상 사용되는 라이브러리로 Vizend Studio에서 프로젝트를 새로 생성할 때 기본으로 포함됩니다. Prologue를 사용하면 Accent 종속성을 포함하고 있어 별도로 추가해 주지 않아도 됩니다.

종속성 설정​

dependencies {
	api("io.vizend:prologue:6.0.1")
}

기본 애플리케이션 설정​

Vizend Prologue는 다양한 설정 기반의 모듈입니다. Spring Auto Configuration 기반으로 구성되어 있어 별도의 Bean 설정을 하지 않아도 동작하고 기본 Properties가 정의되어 있습니다. Spring Boot의 application.yml 에서 기본 설정을 Override하여 변경 적용할 수 있습니다. 다음은 Prologue에 기본 설정된 내용입니다.

  prologue:
    # Prologue 적용 여부를 설정합니다. 만약 false로 설정될 경우 Prologue 보안 설정이
    # 무시 되고 인증 없이 서비스를 호출할 수 있습니다. false는 보통 로컬에서 인증 없이

    permit-all-ant-matchers: # 기본 설정 없음

    # 내부에서 서비스 사이에 호출 시 내부 토큰을 가지고 인증을 대신합니다. 서비스에서 별도의
    # 설정 없이 발급 및 갱신됩니다.
    internal-auth:
      enabled: true
      refresh-interval-seconds: 900 # 15 분

      password: my-internal-secret # Checkpoint 서비스 설정과 동일한 값 필요
      client:
        id: vizend
        secret: my-secret # Checkpoint 토큰 API 호출에 사용할 Basic 인증에 필요
      rest:
        base-url: http://checkpoint:8080
        loopback: false
        base-method: POST
        retry: 3
        request-timeout-seconds: 5
        max-memory-size: 1048576

    # 모든 서버 요청 시점에 사용되고 단일 작업에서의 효율을 위해 일정 시간 캐시됩니다.
    dock:
      enabled: true
      cache-type: ehcache
      cache-ttl-seconds: 1800 # 30 
      cache-refresh-interval-seconds: 900 # 15 분

        base-url: http://metro:8080

        retry: 3

    # 서버와 연계한 바이너리 방식의 이벤트도 사용할 수 있습니다.

      # 있습니다.
      # Ex) servers: 127.0.0.1:4222, 127.0.0.1:5222
      servers: # 기본 설정 없음

      # 붙게 됩니다. 이벤트 발행 뿐 아니라 다른 서비스의 이벤트를 구독할 때 컨슈머 그룹 이름으로도 사용
      # 됩니다. 기본으로 서비스 이름을 지정하게 됩니다.
      # 예) id: io.vizend.${spring.application.name}
      id: # 기본 설정 없음

      # 특정 채널의 이벤트를 수신합니다. 보통 DomainEvent를 수신해 처리하는 용도로 기술하고 목록
      # 형식으로 지정합니다. 채널 이름 앞에 "^" 문자를 붙이면 접미사 없이 채널 이름을 그대로 사용할
      # 수 있습니다.
      # 예) subscriptions: io.vizend.checkpoint, io.vizend.metro, ^custom-topic
      subscriptions: # 기본 설정 없음

      # 이벤트 발행 및 구독할 때 문자 인코딩을 설정합니다.
      encoding: utf-8

      # Set the availability and number of partitions based on the event type.
      event:
        # DataEvent 사용 여부 및 기본 채널의 파티션 개수를 설정합니다. 이 이벤트는 개별 엔티티가
        # 등록, 수정 및 삭제될 때 발생되고 이 이벤트를 사용해 이벤트 소싱 방식의 Replay를 통해
        # 데이터를 재구성 할 때 사용할 수 있습니다. DataEvent는 타 서비스에서 구독해 사용할 수
        # 없습니다.
        data:
          enabled: true
          partition: 2

        # DomainEvent 사용 여부 및 기본 채널의 파티션 개수를 설정합니다. 이 이벤트는 서비스 사이에
        # 이벤트 방식으로 로직을 처리할 때 일반적으로 사용합니다.
        domain:
          enabled: true
          partition: 2

        # RequestEvent 사용 여부 및 기본 채널의 파티션 개수를 설정합니다. Request 및 Response
        # 이벤트를 로깅하고자 할 때 사용하고 REST를 분석하는 목적으로 사용합니다.
        request:
          enabled: true
          partition: 2

        # NamedRequest 이벤트 사용 여부 및 기본 채널의 파티션 개수를 설정합니다.
        # 이 채널 이름은 -domain, -data 등의 접미사가 붙지 않습니다.
        namedChannel:
          enabled: true
          partition: 2

      # 카프카 서비스 설정을 KEY=VALUE 형태의 목록으로 줄 수 있습니다.
      # 예)
      # kafka-configs:
      #   - security.protocol=SASL_SSL
      #   - sasl.mechanism=PLAIN
      #   - sasl.jaas.config=org.apache.kafka.common.security.plain.PlainLoginModule required username='SB3...O6G' password='tNr...g0e';
      kafka-configs: # 기본 설정 없음

      # 스키마 레지스트리 서버 주소를 설정합니다. 여러 서버가 있을 경우 목록으로 지정합니다.
      # 예) http://avro-primary:5000, http://avro-secondary:5000
      schema-registry-urls: # 기본 설정 없음

      # 스키마 레지스트리 서버 값을 설정합니다. KEY=VALUE 형태의 목록으로 줄 수 있습니다.
      # 예)
      # schema-registry-configs:
      #   - basic.auth.user.info=bmV4dHJlZTpwYXNzd29yZA==
      #   - missing.cache.size=20
      schema-registry-configs: # 기본 설정 없음

      # 스키마 레지스트리 서버의 헤더 값을 설정합니다. KEY=VALUE 형태의 목록으로 줄 수 있습니다.
      # 예)
      # schema-registry-http-headers:
      #   - additional-info=actor
      schema-registry-http-headers: # 기본 설정 없음

    # 데이터 추적성을 위한 X-ray 서버 연동 설정 및 정보를 등록합니다.
    xray:
      enabled: true
      stamp-method: rest
      rest:
        base-url: http://xray:8080
      event:
        servers: nats:4222
        subject: io.vizend.xray-server

    # 웹 호출 시 XSS 보안 필터의 세부 설정을 조정합니다. 기본 설정의 HTML 코드 교환시 Escape
    # 처리하고 필요 시 특정 URL을 제외할 수 있습니다. URL은 Ant Matcher를 사용합니다.
    xss-secure:
      enabled: true
      exclude-urls: # 기본 설정 없음
      converter: escape

    # 웹 호출 시 HTTP Method 호출을 제한하여 보안성을 높일 수 있습니다.
    method-secure:
      enabled: true,
      methods: OPTIONS, TRACE, COPY, CONNECT
