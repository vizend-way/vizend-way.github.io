@Slf4j
@Component
@RequiredArgsConstructor
public class FooRestClient {
    //
    private final FooClientProperties properties;

    private RestRequester rest;

    @PostConstruct
    public void initialize() {
        //
        this.rest = new RestRequester(this.properties.getRest());
    }

    public String registerFoo(RegisterFooRequest request) {
        //
        String path = UriComponentsBuilder
                .fromUriString("/restful/foos")
                .toUriString();

        String response = rest.post(path, request);

        if (response == null) {
            throw new IllegalArgumentException(
                    "Cannot register foo, fooRequest = " + request);
        }

        return response;
    }
    
    public List<Foo> findFoos() {
        //
        String path = UriComponentsBuilder
                .fromUriString("/restful/foos")
                .toUriString();

        List<Foo> response = rest.getList(path, Foo.class);

        if (response == null) {
            throw new IllegalArgumentException("Cannot find foos");
        }

        return response;
    }
    
    public Foo findFoo(String id) {
        //
        String path = UriComponentsBuilder
                .fromUriString("/restful/foos/{id}")
                .buildAndExpand(id)
                .toUriString();

        Foo response = rest.get(path, Foo.class);

        if (response == null) {
            throw new IllegalArgumentException("Cannot find foo with id = " + id);
        }

        return response;
    }   
}

@Slf4j
@Component
@RequiredArgsConstructor
public class FooRestClient {
    //
    private final FooClientProperties properties;

    private AsyncReqestRequester rest;

    @PostConstruct
    public void initialize() {
        //
        this.rest = new RestRequester(this.properties.getRest());
    }

    public String registerFoo(RegisterFooRequest request) {
        //
        String path = UriComponentsBuilder
                .fromUriString("/restful/foos")
                .toUriString();

        Mono<String> monoResponse = rest.post(path, request);
        String response = monoResponse.block();

        if (response == null) {
            throw new IllegalArgumentException(
                    "Cannot register foo, fooRequest = " + request);
        }

        return response;
    }

    public List<Foo> findFoos() {
        //
        String path = UriComponentsBuilder
                .fromUriString("/restful/foos")
                .toUriString();

        Mono<List<Foo>> monoResponse = rest.getList(path, Foo.class);
        List<Foo> response = monoResponse.block();

        if (response == null) {
            throw new IllegalArgumentException("Cannot find foos");
        }

        return response;
    }

    public Foo findFoo(String id) {
        //
        String path = UriComponentsBuilder
                .fromUriString("/restful/foos/{id}")
                .buildAndExpand(id)
                .toUriString();

        Mono<Foo> monoResponse = rest.get(path, Foo.class);
        Foo response = monoResponse.block();

        if (response == null) {
            throw new IllegalArgumentException("Cannot find foo with id = " + id);
        }

        return response;
    }
